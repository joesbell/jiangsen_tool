## 把枚举值转为可用数组

```
enum loanTypeEnum {
  INDIVIDUAL_RESIDENCE017 = '个人住房商业贷款',
  INDIVIDUAL_COMMERCIAL = '个人商用房（包括商住两用房）贷款',
  INDIVIDUAL_HAF = '个人住房公积金贷款',
}

class constUtile {
  static geItems(enumType: any) {
    const items = [];
    for (const x in enumType) {
      if (enumType.hasOwnProperty(x)) {
        const obj = {
          value: x,
          name: enumType[x],
        };
        items.push(obj);
      }
    }
    //return [...new Set(items)];  没必要去重了，因为枚举本身的key从定义时就不允许重复
    return items
  }
  static getItemsName(key: string, enumType: any) {
    const arr = this.geItems(enumType);
    for (const n in arr) {
      if (key === arr[n].value) {
        return arr[n].name;
      }
    }
    return '';
  }
}
console.log(constUtile.geItems(loanTypeEnum));
console.log(constUtile.getItemsName('INDIVIDUAL_RESIDENCE017', loanTypeEnum));

```
